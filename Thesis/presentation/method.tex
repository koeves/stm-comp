\begin{frame}{Evaluation}
\onslide<1->{In order to evaluate the performance of the lock-based STM variants, the following metrics are measured:}
\onslide<2->{\begin{itemize}
    \item Time of 10k insertions into the datastructures
    \item Abort rate of the transactional insertion
    \item Speedup per thread compared to the sequential execution
    \item Relative execution time of the transactional API operations
\end{itemize}}
\vspace{0.1cm}
\onslide<3->{The tests were run on 1, 2, 4, 8 and 16 threads respectively on the DAS5 [4]}
\end{frame}
